function boxes(n) { return Math.floor(n / 10); }
function looseC(n) { return n % 10; }
function boxStr(n) {
  const b = boxes(n), l = looseC(n);
  if (n === 0) return '—';
  return b > 0 ? `${b} box${b>1?'es':''} + ${l}` : `${l} cones`;
}

function getStatus(t) {
  if (t.avail === 0) return 'out';
  if (t.avail <= t.min) return 'low';
  return 'ok';
}
function statusBadge(t) {
  const s = getStatus(t);
  if (s === 'out') return '<span class="badge badge-danger">Out of Stock</span>';
  if (s === 'low') return '<span class="badge badge-warn">Low Stock</span>';
  return '<span class="badge badge-ok">In Stock</span>';
}

  document.getElementById('page-' + p).classList.add('active');
  ['dashboard','inventory','lowstock'].forEach((n,i) => {
    if (n === p) document.querySelectorAll('.tab-btn')[i].classList.add('active');
  });
  render();
}

function renderDash() {
  const total = threads.length;
  const totalCones = threads.reduce((s,t) => s + t.avail, 0);
  const totalBoxes = boxes(totalCones);
  const low = threads.filter(t => getStatus(t) === 'low').length;
  const out = threads.filter(t => getStatus(t) === 'out').length;
  document.getElementById('s-total').textContent = total;
  document.getElementById('s-cones').textContent = totalCones;
  document.getElementById('s-boxes').textContent = totalBoxes;
  document.getElementById('s-low').textContent = low;
  document.getElementById('s-out').textContent = out;
  const banner = document.getElementById('alert-banner');
  const issues = low + out;
  if (issues > 0) {
    banner.classList.remove('hidden');
    document.getElementById('alert-text').textContent = `${issues} thread${issues>1?'s':''} need attention — ${out} out of stock, ${low} running low.`;
  } else { banner.classList.add('hidden'); }
  const tbody = document.getElementById('dash-tbody');
  if (!threads.length) { tbody.innerHTML = `<tr><td colspan="7"><div class="empty"><div class="empty-icon">🧵</div><p>No threads yet.</p></div></td></tr>`; return; }
  tbody.innerHTML = threads.map(t => `
    <tr>
      <td><div class="color-cell"><span class="color-dot" style="background:${t.hex}"></span></div></td>
      <td><code>${t.code}</code></td>
      <td style="font-weight:500">${t.name}</td>
      <td><span class="badge badge-type">${t.type}</span></td>
      <td class="num">${t.avail}</td>
      <td><span class="box-badge">📦 ${boxStr(t.avail)}</span></td>
      <td>${statusBadge(t)}</td>
    </tr>`).join('');
}

// LOW STOCK
// ══════════════════════════════
function renderLowStock() {
  const issues = threads.filter(t => getStatus(t) !== 'ok').sort((a,b) => a.avail - b.avail);
  document.getElementById('ls-count').textContent = issues.length ? `${issues.length} item${issues.length>1?'s':''} need attention` : '';
  const list = document.getElementById('ls-list');
  if (!issues.length) { list.innerHTML = `<div class="empty"><div class="empty-icon">✅</div><p>All threads are well stocked!</p></div>`; return; }
  list.innerHTML = issues.map(t => {
    const s = getStatus(t);
    const needed = Math.max(0, t.min - t.avail);
    const boxesNeeded = Math.ceil(needed / 10);
    return `<div class="ls-item ${s==='out'?'critical':'warning'}">
      <span class="color-dot" style="background:${t.hex};width:32px;height:32px;"></span>
      <div class="ls-info">
        <div class="ls-name">${t.name} <code>${t.code}</code></div>
        <div class="ls-meta">${t.type} · ${t.location||'—'}</div>
        <div style="margin-top:5px;display:flex;align-items:center;gap:8px;flex-wrap:wrap">
          ${statusBadge(t)}
          ${needed>0?`<span style="font-size:12px;color:var(--muted)">Order <b>${needed}</b> more cones = <b>${boxesNeeded} box${boxesNeeded>1?'es':''}</b></span>`:''}
        </div>
      </div>
      <div class="ls-right">
        <div class="ls-cones" style="color:${s==='out'?'var(--danger)':'var(--warn)'}">${t.avail}</div>
        <div class="ls-min">min: ${t.min}</div>
        <span class="box-badge" style="margin-top:4px;display:inline-block">📦 ${boxStr(t.avail)}</span>
      </div>
    </div>`;
  }).join('');
}

let jollyMatch = null;
function jollyLookup() {
  const raw = document.getElementById('jolly-input').value.trim().toUpperCase();
  // normalise: "415" → "J-415", "J415" → "J-415", "J-415" → "J-415"
  let code = raw;
  if (/^\d+$/.test(code)) code = 'J-' + code;
  else if (/^J\d+$/.test(code)) code = 'J-' + code.slice(1);

  const hex = JOLLY[code];
  const name = JOLLY_NAMES[code];
  const resultEl = document.getElementById('jolly-result');
  const previewEl = document.getElementById('jolly-preview');
  const infoEl = document.getElementById('jolly-info');

  if (hex) {
    jollyMatch = { code, hex, name: name || code };
    previewEl.style.background = hex;
    infoEl.textContent = `${code} — ${name || 'Jolly Thread'} (${hex})`;
    resultEl.classList.remove('hidden');
  } else {
    jollyMatch = null;
    resultEl.classList.add('hidden');
  }
}

function applyJolly() {
  if (!jollyMatch) { toast('Code not found in Jolly catalogue'); return; }
  document.getElementById('f-code').value = jollyMatch.code;
  document.getElementById('f-name').value = jollyMatch.name;
  document.getElementById('f-hex').value = jollyMatch.hex;
  syncColor();
  toast(`Applied ${jollyMatch.code} — ${jollyMatch.name}`);
}

function openModal(id = null) {
  editingId = id;
  document.getElementById('modal-title').textContent = id ? 'Edit Thread' : 'Add Thread';
  document.getElementById('jolly-input').value = '';
  document.getElementById('jolly-result').classList.add('hidden');
  jollyMatch = null;

  if (id) {
    const t = threads.find(x => x.id === id);
    document.getElementById('f-code').value = t.code;
    document.getElementById('f-name').value = t.name;
    document.getElementById('f-hex').value = t.hex;
    document.getElementById('f-type').value = t.type;
    document.getElementById('f-brand').value = t.brand || '';
    document.getElementById('f-avail').value = t.avail;
    document.getElementById('f-reserved').value = t.reserved;
    document.getElementById('f-min').value = t.min;
    document.getElementById('f-location').value = t.location || '';
    document.getElementById('jolly-input').value = t.code;
    jollyLookup();
    syncColor();
    updateBoxCalc();
  } else {
    ['f-code','f-name','f-hex','f-brand','f-location'].forEach(i => document.getElementById(i).value = '');
    document.getElementById('f-avail').value = 0;
    document.getElementById('f-reserved').value = 0;
    document.getElementById('f-min').value = 10;
    document.getElementById('color-preview').style.background = '#ccc';
    document.getElementById('box-helper').classList.add('hidden');
  }
  document.getElementById('overlay').classList.remove('hidden');
}

function closeModal() { document.getElementById('overlay').classList.add('hidden'); editingId = null; }
function overlayBg(e) { if (e.target === document.getElementById('overlay')) closeModal(); }

function syncColor() {
  const hex = document.getElementById('f-hex').value;
  document.getElementById('color-preview').style.background = hex || '#ccc';
}

function updateBoxCalc() {
  const n = parseInt(document.getElementById('f-avail').value) || 0;
  const helper = document.getElementById('box-helper');
  if (n >= 0) {
    const b = boxes(n), l = looseC(n);
    helper.innerHTML = `📦 ${n} cones = <b>${b} full box${b!==1?'es':''}</b> + ${l} loose cone${l!==1?'s':''} (10 cones/box)`;
    helper.classList.remove('hidden');
  } else { helper.classList.add('hidden'); }
}

function saveThread() {
  const code = document.getElementById('f-code').value.trim();
  const name = document.getElementById('f-name').value.trim();
  if (!code) { toast('Jolly Code is required'); return; }
  const data = {
    code,
    name: name || code,
    hex: document.getElementById('f-hex').value.trim() || '#888888',
    type: document.getElementById('f-type').value,
    brand: document.getElementById('f-brand').value.trim() || 'Jolly Group',
    avail: parseInt(document.getElementById('f-avail').value) || 0,
    reserved: parseInt(document.getElementById('f-reserved').value) || 0,
    min: parseInt(document.getElementById('f-min').value) || 0,
    location: document.getElementById('f-location').value.trim(),
  };
  if (editingId) {
    const idx = threads.findIndex(t => t.id === editingId);
    threads[idx] = { ...threads[idx], ...data };
    toast('Updated ✓');
  } else {
    data.id = Date.now();
    threads.push(data);
    toast('Thread added ✓');
  }
  save(); closeModal(); render();
}

function delThread(id) {
  if (!confirm('Delete this thread?')) return;
  threads = threads.filter(t => t.id !== id);
  save(); render(); toast('Deleted');
}

  const el = document.getElementById('toast');
  el.textContent = msg; el.classList.add('show');
  clearTimeout(_tt); _tt = setTimeout(() => el.classList.remove('show'), 2500);
}
